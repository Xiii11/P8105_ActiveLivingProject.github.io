transfer datatype to numeric, filter by borough, using total number for each commuting mode

borough_df = 
  commuting_df  |> 
  filter(geo_type_desc == "Borough") |> 
  mutate(
    across(c(bicycle_number, car_truck_or_van_number, public_transportation_number, walked_number),
                ~as.numeric(gsub(",", "", .))))
